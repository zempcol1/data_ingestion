                      height=400,

url = 'http://overpass-api.de/api/interpreter'

query = f"""
        [out:json];
        area["ISO3166-1"="CH"][admin_level=2];
        node ["leisure"="stadium"](area);
        out;"""

r = requests.get(url, params={'data': query})
